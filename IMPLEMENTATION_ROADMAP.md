```typescript

POST   /api/inventory/alerts
GET    /api/inventory/alerts/:merchantId
PUT    /api/inventory/alerts/:id
DELETE /api/inventory/alerts/:id

// Smart Contracts API
POST   /api/contracts
GET    /api/contracts/:userId
PUT    /api/contracts/:id/renew
GET    /api/contracts/:id/performance

// RBAC API
GET    /api/roles
POST   /api/roles
```
